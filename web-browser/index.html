Browser Utils #

浏览器的一些工具函数，挺好用的，至少我认为是这样。

Install #

bash

npm i @hairy/browser-utils

Type Declarations #

typescript
interface OpenFilePickerOptions {
  /**
   * 是否多选文件
   */
  multiple?: boolean
  /**
   * 选择文件的默认格式
   */
  accept?: string
}
/**
 * 选择多个文件
 * @param option.multiple 是否多选
 * @param option.accept 文件类型(accept)
 */
function showOpenFilePicker(option?: OpenFilePickerOptions): Promise<File[]>
interface OpenImagePickerOptions {
  /**
   * 是否多选图片
   */
  multiple?: boolean
}
/**
 * 选择多个图片
 * @param multiple 是否多选
 */
function showOpenImagePicker(options?: OpenImagePickerOptions): Promise<File[]>
/** @deprecated use chooseFile */
const selectImages: typeof showOpenImagePicker
/**
 * 生成 blob|string 文件，并下载
 * @param data blob 数据，或者字符串
 * @param name 文件名称
 */
function downloadBlobFile(data: Blob | string, name: string): void
/**
 * 下载网络文件
 * @param url 下载地址
 * @param name 文件名称
 */
function downloadNetworkFile(url: string, name?: string): void
type ReaderType =
  | 'readAsArrayBuffer'
  | 'readAsBinaryString'
  | 'readAsDataURL'
  | 'readAsText'
/**
 * 读取 File 文件
 * @param formType 转换形式
 * @param file 文件
 */
function readFileReader<T extends ReaderType>(
  formType: T,
  file: File
): Promise<T extends 'readAsArrayBuffer' ? ArrayBuffer : string>

function redirectTo(url: string, target?: string): void
function callPhoneNumber(phoneNumber: string): void

export {
  OpenFilePickerOptions,
  OpenImagePickerOptions,
  ReaderType,
  callPhoneNumber,
  downloadBlobFile,
  downloadNetworkFile,
  readFileReader,
  redirectTo,
  selectImages,
  showOpenFilePicker,
  showOpenImagePicker,
}
